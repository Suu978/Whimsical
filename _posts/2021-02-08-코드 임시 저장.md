package day_10;

public class Ex1 {
	
	static int abs (int value)
	{
		return value > 0 ? value : -value;
	}
	
	public static void main(String[] arge)
	{
		int value = 5;
		System.out.println(value+"의 절대값 : "+abs(value));
		value = -10;
		System.out.println(value+"의 절대값 : "+abs(value));
	}
}
