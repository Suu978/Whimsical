package day_9;

class Human {
	private String name;
	private int haight;
	private int weight;
		
	Human(){};
	Human(String name, int haight, int weight) {
		this.name = name;
		this.haight = haight;
		this.haight = weight;
	}
	String getName() {
		return this.name;
	}
	int getHaight() {
		return this.haight;
	}
	int getWeight() {
		return this.weight;
	}
}

class HumanTast {
	
	public static void name(String[] args) {
		
		Human hodu = new Human("호두", 50, 5);
		Human ddosik = new Human("또식", 30, 3);
		
		System.out.println("이름 : " + hodu.getName());
		System.out.println("키 : " + hodu.getHaight() + "cm");
		System.out.println("몸무게 : " + hodu.getWeight() + "kg");
		System.out.println();

		System.out.println("이름 : " + ddosik.getName());
		System.out.println("키 : " + ddosik.getHaight() + "cm");
		System.out.println("몸무게 : " + ddosik.getWeight() + "kg");
		System.out.println();
	}	
}
